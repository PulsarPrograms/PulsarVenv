help - вывести эту справку
out "text" - вывести текст на экран
pulsar info - показать информацию о текущем профиле

setrule warnings [true/false] - включить/выключить предупреждения
setrule locale [язык] - установить язык интерфейса
setrule theme [тема] - изменить тему оформления
setrule betafunc [true/false] - включить/выключить бета-функции
setrule tcolor [цвет] - изменить цвет терминала
setrule showpath [true/false] - показывать/скрыть текущий путь

config show - показать текущую конфигурацию

|BETA| alias show - показать все алиасы
|BETA| alias create [алиас] [команда] - создать новый алиас
|BETA| alias remove [алиас] - удалить алиас

log write [текст] [true/false] - записать в лог (с временем/без)
log show - показать содержимое лога
log clear - очистить лог (с подтверждением)

calc - Доступ к калькулятору системы
script start [путь] - выполнить скрипт
exit - Выход из программы
whoami - Показать текущий профиль
clear - Очистка экрана терминала (Обновляет систему в соответсвии с конфигом)


cd .. - перейти в родительскую директорию
cd .... - перейти на два уровня вверх
cd ~LAST~ - вернуться к предыдущей директории
cd ~ - перейти в домашнюю директорию
cd (без аргументов) - перейти в домашнюю директорию


betcd [имя_директории] - умный поиск директории в подриекториях
pwd - показать текущий путь

mkdir [путь] - создать новую директорию
mkcd [путь] - создать директорию и перейти в неё
rmdir [путь] - удалить директорию со всеми файлами
ls [путь] - список содержимого указанной директории
ls  - список содержимого текущий директории
tree - показать дерево директорий 

crfile [путь] - создать новый файл
crwfile [путь] [текст] - создать файл и записать в него текст (с поддержкой escape-последовательностей \n, \t)
more [путь] - просмотреть содержимое файла
writef [путь] [текст] - дописать текст в файл (с поддержкой escape-последовательностей)
rm [путь] - удалить файл

fsdown [номер]* - перейти в указанную поддиректорию *(По умолчанию = 1)


profile add [имя] - создать новый профиль
profile swap [имя] - переключиться на другой профиль
profile list - показать список всех профилей (Текущий профиль выделяется цветом)
profile setlike [имя] - установить профиль для автоматического входа при старте
